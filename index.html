const SHEET_DEL = '配送先';
const SHEET_RTE = 'ルート履歴';

function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const type = e.parameter.type || 'deliveries';
  if (type === 'routes') {
    const sheet = ss.getSheetByName(SHEET_RTE);
    const rows = sheet.getDataRange().getValues();
    const headers = rows[0];
    const data = rows.slice(1).filter(r => r[0]).map(row => {
      const obj = {};
      headers.forEach((h, i) => obj[h] = row[i]);
      try { obj.stops = JSON.parse(obj.stops); } catch(e) { obj.stops = []; }
      return obj;
    });
    return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
  }
  const sheet = ss.getSheetByName(SHEET_DEL);
  const rows = sheet.getDataRange().getValues();
  const headers = rows[0];
  const data = rows.slice(1).filter(r => r[0]).map(row => {
    const obj = {};
    headers.forEach((h, i) => obj[h] = row[i]);
    obj.tags = obj.tags ? String(obj.tags).split(',').map(t => t.trim()).filter(Boolean) : [];
    return obj;
  });
  return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const payload = JSON.parse(e.postData.contents);
  if (payload.action === 'saveDeliveries') {
    const sheet = ss.getSheetByName(SHEET_DEL);
    const lastRow = sheet.getLastRow();
    if (lastRow > 1) sheet.deleteRows(2, lastRow - 1);
    payload.data.forEach(d => sheet.appendRow([d.id, d.name, d.addr, d.contact||'', d.note||'', (d.tags||[]).join(',')]));
  } else if (payload.action === 'saveRoute') {
    const sheet = ss.getSheetByName(SHEET_RTE);
    if (payload.update) {
      const rows = sheet.getDataRange().getValues();
      for (let i = 1; i < rows.length; i++) {
        if (String(rows[i][0]) === String(payload.route.id)) {
          sheet.getRange(i+1,5).setValue(JSON.stringify(payload.route.stops));
          break;
        }
      }
    } else {
      sheet.appendRow([payload.route.id, payload.route.date, payload.route.user, payload.route.label, JSON.stringify(payload.route.stops)]);
    }
  } else if (payload.action === 'deleteRoute') {
    const sheet = ss.getSheetByName(SHEET_RTE);
    const rows = sheet.getDataRange().getValues();
    for (let i = rows.length - 1; i >= 1; i--) {
      if (String(rows[i][0]) === String(payload.id)) { sheet.deleteRow(i + 1); break; }
    }
  }
  return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
}
